library(torch)

reparameterize <- function(mu, logvar, sampling = TRUE, use_cuda) {
  # for X ~ N(mu, sigma^2), can rewrite as X = mu + sigma * Z
  # "reparam trick" often used to preserve gradient on mu and sigma
  # Last modified 2024/07/16
  # modified 2025/10/27 --- made use_cuda argument extraneous 
  #   - mucked up computation when moving model off cuda / to cuda 
  #     because used stored info to decide whether was cuda or not.
  #   - made extraneous for compatibility with already trained models

  if (sampling) {
    std <- logvar$mul(0.5)$exp_()
    if (mu$is_cuda) {
      eps <- torch_randn(std$size(), device = "cuda", requires_grad = TRUE)
    } else {
      eps <- torch_randn(std$size(), device = "cpu", requires_grad = TRUE)
    }
    return(mu$add(eps$mul(std)))
  } else {
    return(mu)
  }
}



KL_lognorm_gamma <- function(mu, logvar, a = 1/2, b = 1){
  # mu and logvar must be torch_tensors
  # computes KL(logNormal(mu, sig^2)  ||  Gamma(a, b))
  # used for s_a, alpha_i
  # corrected; full KL:
  #   - 1/2 * (log(2) + log(pi) + log(sig^2) + 1)
  #   + log(gamma(a)) + a*(log(b) - mu)
  #   + 1/b * exp(mu + sig^2 / 2)
  # want negative KL
  # expr1 <- -0.5 * (log(2) + log(pi) + logvar + 1)
  # expr2 <- lgamma(a) - (mu$add(-log(b)))$mul(a)
  # expr3 <- (mu$add(  (logvar$exp())$mul(1/2)  ))$exp() / b
  inner1 <- 1 + log(2) + log(pi) + logvar
  inner2 <- mu + logvar$exp()$mul(1/2)
  inner1$mul(-0.5) + lgamma(a) + a*log(b) - mu$mul(a) + inner2$exp() / b
}



KL_lognorm_IG <- function(mu, logvar, a = 1/2, b = 1){
  # for s_b, beta_i
  # i.e. KL(LogNormal q || Inverse-Gamma p)
  # corrected; full KL:
  #   - 1/2 * (log(2) + log(pi) + log(sig^2) + 1)
  #   + log(gamma(a)) + a*(mu - log(b))
  #   + b * exp(-mu + sig^2 / 2)
  # same as KL(LogNormal q || Gamma p) if replace mu with -mu, b with 1/b
  KL_lognorm_gamma(mu = -mu, logvar = logvar, a = a, b = 1/b)
}

E_lognorm <- function(mu, logvar){
  # expectation of X ~ logNormal(mu, var)
  # exp(mu + (sig^2)/2)
  # ln_mu <- mu$add( logvar$exp()$mul(0.5) )
  # return(ln_mu$exp())
  # alternatively:
  (mu$add(  logvar$exp()$mul(0.5) ))$exp()
}


V_lognorm <- function(mu, logvar){
  # variance of X ~ logNormal(mu, var)
  # (exp(var) - 1) * exp(2*mu + var)
  expr1 <- logvar$exp()$exp() - 1
  expr2 <- (mu$mul(2))$add(logvar$exp())
  return(expr1$mul(expr2$exp()))
}


E_sqrt_lognorm <- function(mu, logvar){
  # if X ~ LN(mu, var), then X = exp(Y) with Y~N(mu, var)
  # The sqrt(X) = sqrt(exp^Y) = exp^{Y/2} and Y/2 ~ N(mu/2, var/4)
  E_lognorm(mu/2, logvar - log(4))
}

V_sqrt_lognorm <- function(mu, logvar){
  V_lognorm(mu/2, logvar - log(4))
}

mu_sqrt_prod_LN <- function(mu_1, mu_2){
  # \frac{mu_1 + mu_2}{2}
  (mu_1$add(mu_2))$mul(1/2)
}

logvar_sqrt_prod_LN <- function(logvar_1, logvar_2){
  # \frac{sig^2_1 + sig^2_2}{4}
  ((logvar_1$exp())$add(logvar_2$exp()))$log() - log(4)
}

log_dropout <- function(hs_layer, type = "local"){
  # calculates dropout rates based on :
  # type == "local":    ztilde = sqrt(atilde btilde)
  # type == "global":    s = sqrt(sa sb)
  # type == "marginal":    z = ztilde * s
  
  if (type == "local"){
    logvar_sum <- hs_layer$atilde_logvar + hs_layer$btilde_logvar
  } else if (type == "global"){
    logvar_sum <- hs_layer$sa_logvar + hs_layer$sb_logvar
  } else if (type == "marginal"){
    logvar_sum <- hs_layer$atilde_logvar + hs_layer$btilde_logvar + hs_layer$sa_logvar + hs_layer$sb_logvar
  }
  
  type_var <- exp(logvar_sum - log(4))
  log_dropout <- log(exp(type_var) - 1)
  return(log_dropout)
}

  get_kl = function() {
    
    # KL(q(s_a) || p(s_a));   logNormal || Gamma
    kl_sa <- KL_lognorm_gamma(mu = self$sa_mu, logvar = self$sa_logvar, a = 1/2, b = self$tau_0)

    # KL(q(btilde) || p(btilde));   logNormal || invGamma

    # KL(q(w|z) || p(w|z))
    kl_w_z <- torch_sum(
      0.5 * (-self$weight_logvar + self$weight_logvar$exp() + self$weight_mu$pow(2) - 1)
    )
    
    # KL for bias term

## 🔴 Correctness Issues

### 1. In-Place `$exp_()` in `reparameterize` Can Corrupt Gradients

```r
std <- logvar$mul(0.5)$exp_()   # ← in-place on a graph tensor
```

Although `$mul(0.5)` creates a new tensor, `$exp_()` modifies it **in-place** while it's part of the autograd computation graph. This can cause silent gradient errors or an outright error like *"a variable needed for gradient computation has been modified by an inplace operation."* Fix:

```r
std <- logvar$mul(0.5)$exp()    # out-of-place
```

### 2. `clip_variances()` Orphans Parameters from the Optimizer

```r
clip_variances = function() {
  if (!is.null(self$clip_var)) {
    # These create BRAND NEW nn_parameter objects:
    self$weight_logvar <- nn_parameter(self$weight_logvar$clamp(max = log(self$clip_var)))
    self$bias_logvar   <- nn_parameter(self$bias_logvar$clamp(max = log(self$clip_var)))
  }
}
```

Every call constructs new `nn_parameter` objects. The optimizer still holds references to the **old** parameters, so gradients on the new parameters are never applied. If you call this during training, weight and bias variances effectively stop updating. Fix with in-place clamping:

```r
clip_variances = function() {
  if (!is.null(self$clip_var)) {
    with_no_grad({
      self$weight_logvar$clamp_(max = log(self$clip_var))
      self$bias_logvar$clamp_(max = log(self$clip_var))
    })
  }
}
```

### 3. τ₀ Scale Is Off by a Square Root

You decompose τ² = s_a · s_b where s_a ~ Gamma(½, scale = b) and s_b ~ InvGamma(½, scale = 1). This yields:

$$\tau^2 = s_a \cdot s_b \sim b \cdot \text{Gamma}(\tfrac{1}{2}, 1) \cdot \text{InvGamma}(\tfrac{1}{2}, 1) = b \cdot C^+(0,1)^2$$

So τ ~ C⁺(0, √b). In `get_kl` you pass `b = self$tau_0`:

```r
kl_sa <- KL_lognorm_gamma(mu = self$sa_mu, logvar = self$sa_logvar, a = 1/2, b = self$tau_0)
```

If you intend τ ~ C⁺(0, τ₀), you need **`b = tau_0^2`**, not `b = tau_0`. With the default `tau_0 = 1` this is invisible, but for any other value the effective prior scale is √τ₀ instead of τ₀.

## 🟠 Convergence / Performance Issues

### 4. Sampling 4 Auxiliary Variables Separately — High Gradient Variance *(likely your biggest convergence issue)*

Your forward pass draws **four independent** noise samples (for log_atilde, log_btilde, log_sa, log_sb) and combines them:

```r

log_btilde <- reparameterize(mu = self$btilde_mu, logvar = self$btilde_logvar, ...)

```

Each independent noise source adds variance to the ELBO gradient estimate. Since log(z) is a linear combination of four independent normals, you can sample it **directly** as a single Gaussian — the commented-out code already does this:

```r

  mu = (self$atilde_mu + self$btilde_mu + self$sa_mu + self$sb_mu) / 2,
  logvar = torch_log(
    self$atilde_logvar$exp() + self$btilde_logvar$exp() +
    self$sa_logvar$exp() + self$sb_logvar$exp()
  ) - log(4),
  use_cuda = self$use_cuda
)

```

> **Note:** the commented-out version in your code has the same log-variance summation bug described in Issue #8 below — use `torch_log(sum of exp(...))`, not a plain sum of logvars.

This single-sample approach gives **identical** marginal distributions but with **much lower** gradient variance. This is very likely the dominant cause of your slow convergence.

### 5. `requires_grad = TRUE` on Noise Tensor `eps`

```r

```

The noise `eps` is not a model parameter — gradients should flow through `mu` and `std`, not `eps`. Setting `requires_grad = TRUE` on it forces the autograd engine to track unnecessary graph nodes, wasting memory and slightly slowing every backward pass. Fix:

```r

# requires_grad defaults to FALSE — this is correct
```

(Also note: using `mu$device` is cleaner than the if/else branch.)

### 6. Bias Prior N(0, 1) May Be Overly Restrictive

The KL for the bias is:

```r
kl_bias <- torch_sum(0.5 * (-self$bias_logvar + self$bias_logvar$exp() + self$bias_mu$pow(2) - 1))
```

This is KL(q(b) ‖ N(0, 1)). An N(0, 1) prior actively regularizes biases toward zero, which can fight the model's ability to shift activations. Consider a broader prior (e.g., N(0, σ² = 100)) and adjust the KL term accordingly:

```r
# KL(N(mu, sig2) || N(0, prior_var)):
prior_var <- 100
kl_bias <- torch_sum(0.5 * (
  log(prior_var) - self$bias_logvar +
  (self$bias_logvar$exp() + self$bias_mu$pow(2)) / prior_var - 1
))
```

## 🟡 Numerical Stability Issues

### 7. No Overflow Guard on `z = exp(...)`

```r

```

If the log-shrinkage variables produce large positive values, this overflows to `Inf`, and subsequent operations yield `NaN`. Add a clamp:

```r
z <- (log_s + log_ztilde)$clamp(max = 30)$exp()   # exp(30) ≈ 1e13, safe
```

### 8. `var_activations$log()` Has No Floor

```r
var_activations <- nnf_linear(input = xz$pow(2), weight = self$weight_logvar$exp(), ...)
# var_activations is guaranteed positive, but can be tiny

```

Extremely small variances produce very large negative log-values, which can destabilize the reparameterization. Safeguard with:

```r
logvar = var_activations$clamp(min = 1e-16)$log()
```

## 🔵 Post-Hoc Analysis Bugs (Won't Affect Training, But Give Wrong Results)

### 9. Log-Variance Summation Bug — Product Instead of Sum

In `compute_posterior_param`, `get_Eztilde_i`, `get_Vztilde_i`, and the standalone `log_dropout`, you combine log-variances by **adding** them:

```r

```

Since `logvar = log(σ²)`, adding log-variances gives log(∏ σ²ᵢ), i.e., the **product** of variances. But the variance of a sum of independent normals is the **sum** of variances, so you need:

```r
logvar = torch_log(
  self$atilde_logvar$exp() + self$btilde_logvar$exp() +
  self$sa_logvar$exp() + self$sb_logvar$exp()
) - log(4)
```

This same bug pattern applies to:
- `get_Eztilde_i` and `get_Vztilde_i` (2 terms)
- `log_dropout` (2 or 4 terms depending on type)

Notably, **`get_dropout_rates` does this correctly** (it exponentiates before adding), so the two dropout functions are mutually inconsistent.

binary_err_mat <- function(est, tru){
  # returns 4-row matrix of FP, TP, FN, TN
  FP <- est - tru == 1
  TP <- est + tru == 2
  FN <- est - tru == -1
  TN <- abs(tru) + abs(est) == 0
  return(rbind(FP, TP, FN, TN))
}

binary_err <- function(est, tru){
  # returns FP, TP, FN, TN as percentage of all decisions
  rowSums(binary_err_mat(est, tru)) / length(tru)  
}

binary_err_rate <- function(est, tru){
  # returns FP, TP, FN, TN rates
  decision_counts <- rowSums(binary_err_mat(est, tru))
  actual_pos <- decision_counts[2] + decision_counts[3]
  actual_neg <- decision_counts[1] + decision_counts[4]
  
  denom <- c(actual_neg, actual_pos, actual_pos, actual_neg)
  # in case no positives or negatives predicted
  denom <- ifelse(denom == 0, 1, denom) 
  decision_counts / denom
}

BFDR <- function(dropout_probs, eta){
  delta_vec <- 1 * (dropout_probs <= eta)
  if (sum(delta_vec) == 0){ 
    warning("no included variables; returning BFDR = 0") 
    bfdr <- 0
  } else {
    bfdr <- sum((dropout_probs) * delta_vec) / sum(delta_vec)
  }
  return(
    list(
      "delta_i" = delta_vec,
      "bfdr" = bfdr
    )
  )
}

FDR <- function(delta_vec, true_gam = c(rep(1, 4), rep(0, 100))){
  if (sum(delta_vec) == 0){
    return(0)
  } else {
    return(sum((delta_vec - true_gam) == 1) / sum(delta_vec))
  }
}

BFDR_eta_search <- function(dropout_probs, max_rate = 0.05){
  a_sort <- sort(dropout_probs)
  bfdrs <- sapply(a_sort, function(X) BFDR(dropout_probs, eta = X)$bfdr)
  inds <- which(bfdrs <= max_rate)
  if (length(inds)==0){
    warning("no threshold found, returning eta = 0")
    return(0)
  } else {
    a_sort[max(inds)]
  }
}

err_from_dropout <- function(
    dropout_vec, 
    max_bfdr = 0.01, 
    true_gam = c(rep(1, 4), rep(0, 100))
){
  eta <- BFDR_eta_search(dropout_vec, max_rate = max_bfdr)
  bfdr <- BFDR(dropout_vec, eta)$bfdr
  delta_i <- BFDR(dropout_vec, eta)$delta_i
  bin_err <- binary_err_rate(est = delta_i, tru = true_gam)
  fdr <- FDR(delta_vec = delta_i, true_gam = true_gam)
  c("fdr" = fdr, "bfdr" = bfdr, bin_err)
}

get_ztil_sq <- function(nn_model_layer, ln_fcn = ln_mode){
  ztil_params <- get_ztil_params(nn_model_layer)
  ztil_sq <- ln_fcn(
    ztil_params$at + ztil_params$bt, 
    exp(ztil_params$at_lvar) + exp(ztil_params$bt_lvar)
  )
  return(ztil_sq)
}

get_zsq <- function(nn_model_layer, ln_fcn = ln_mode){
  s_sq <- get_s_sq(nn_model_layer, ln_fcn)
  ztil_sq <- get_ztil_sq(nn_model_layer, ln_fcn)
  return(s_sq * ztil_sq)
}



get_kappas <- function(nn_model_layer, type = "global"){
  ztil_sq <- get_ztil_sq(nn_model_layer)
  if (type == "global"){
    s_sq <- get_s_sq(nn_model_layer)
    kappas <- 1 / ( 1 + s_sq*ztil_sq)
  } else if (type == "local"){
    kappas <- 1 / ( 1 + ztil_sq)
  } else {
    warning("type must be global or local")
  }
  return(kappas)
}

get_wtil_params <- function(nn_model_layer){
  wtil_lvar <- as_array(nn_model_layer$weight_logvar)
  wtil_mu <- as_array(nn_model_layer$weight_mu)
  return(
    list(
      "wtil_lvar" = wtil_lvar,
      "wtil_mu" = wtil_mu
    )
  )
}

  z_sq <- get_s_sq(nn_model_layer) * get_ztil_sq(nn_model_layer)

# PRIOR CALIBRATION ----

## tau0_PV ----
tau0_PV <- function(p_0, d, sig = 1, n){
  # Piironen & Vehtari 2017 suggest tau_0 = p_0 / (d - p_0) * sig / sqrt(n)
  # where p_0 = prior estimate of number of nonzero betas, d = total number of covs
  p_0 / (d - p_0) * sig / sqrt(n)
}


# PARAM CORRECTIONS ----

## m_eff ----
m_eff <- function(nn_layer){
  k <- get_kappas(nn_layer)
  sum(1 - k)
}

## frob(enius norm) ----
frob <- function(mat){
  sqrt(sum(mat^2))
}


## tau_correction ----
tau_correction <- function(nn_mod){
  d_1 <- length(get_kappas(nn_mod$fc2))
  m_2 <- m_eff(nn_layer = nn_mod$fc2)
  
  return(sqrt(d_1 / m_2))
}


## get_kappas_taucorrected ----
get_kappas_taucorrected <- function(nn_mod, ln_fcn = ln_mode){
  zsq_1 <- get_zsq(nn_mod$fc1, ln_fcn)
  tau_correction_factor <- tau_correction(nn_mod)
  return((1 + zsq_1*tau_correction_factor^2)^(-1))
}


## get_kappas_frobcorrected ----
get_kappas_frobcorrected <- function(nn_mod, ln_fcn = ln_mode){
  
  z1_c <- sqrt(get_zsq(nn_mod$fc1))
  for (l in 2:length(nn_mod$children)){
    z_l <- sqrt(get_zsq(nn_mod$children[[l]]))
    w_l_mu <- get_wtil_params(nn_mod$children[[l]])$wtil_mu
    z1_c <- z1_c * frob(diag(z_l) %*% t(w_l_mu))
  }
  
  return((1 + z1_c^2)^(-1))
}

# GENERATE PREDS ----
get_nn_mod_Ey <- function(nn_mod, X, ln_fcn = ln_mode){
  num_layers <- length(nn_mod$children)
  # select cuda or cpu
  use_cuda <- nn_mod$children[[1]]$sa_mu$is_cuda
  cuda_or_cpu <-dev_select(use_cuda)
  if (use_cuda){
    input <- X$cuda()
  } else {
    input <- X$cpu()
  }

  for (nn_layer in 1:num_layers){
    
    ztilde <- ln_fcn(
      mu = as_array(nn_mod$children[[nn_layer]]$atilde_mu + nn_mod$children[[nn_layer]]$btilde_mu)/2,
      var = exp(as_array(nn_mod$children[[nn_layer]]$atilde_logvar + nn_mod$children[[nn_layer]]$btilde_logvar))/4
    )
    
    s <- ln_fcn(
      mu = as_array(nn_mod$children[[nn_layer]]$sa_mu + nn_mod$children[[nn_layer]]$sb_mu)/2,
      var = exp(as_array(nn_mod$children[[nn_layer]]$sa_logvar + nn_mod$children[[nn_layer]]$sb_logvar))/4
    )
    
    z <- torch_tensor(ztilde*s, device = cuda_or_cpu)
    
    Xz <- input*z
    
    mu_activations <- nnf_linear(
      input = Xz, 
      weight = nn_mod$children[[nn_layer]]$weight_mu, 
      bias = nn_mod$children[[nn_layer]]$bias_mu
    )
    
    if (nn_layer < num_layers){
      # input for next layer
      input <- nnf_relu(mu_activations)
    }
  }
  
  return(mu_activations)
}



# FOR LM() ----
calc_lm_stats <- function(lm_fit, true_coefs, alpha = 0.05){
  beta_hat <- summary(lm_fit)$coef[-1, 1]
  binary_err <- binary_err_rate(
    est = summary(lm_fit)$coef[-1, 4] < alpha, 
    tru = true_coefs != 0)
  fit_mse <- mean(lm_fit$residuals^2)
  coef_mse <- mean((beta_hat - true_coefs)^2)
  list(
    "binary_err" = binary_err,
    "fit_mse" = fit_mse,
    "coef_mse" = coef_mse
  )
}

get_lm_stats <- function(simdat, alpha = 0.05){
  lm_df <- data.frame(
    "y" = as_array(simdat$y), 
    "x" = as_array(simdat$x)
  )
  if (simdat$d_in > simdat$n_obs){
    lm_df <- lm_df[, 1:(ceiling(simdat$n_obs/2)+1)]
  }
  
  lm_fit <- lm(y ~ ., lm_df)
  if (length(simdat$true_coefs) >= simdat$n_obs){
    warning("p >= n; (p - n) + floor(n/2) spurious covariates eliminated to accomodate lm")
    calc_lm_stats(
      lm_fit = lm_fit, 
      true_coefs = simdat$true_coefs[1:ceiling(simdat$n_obs/2)], 
      alpha = alpha
    )
  } else {
    calc_lm_stats(lm_fit = lm_fit, true_coefs = simdat$true_coefs, alpha = alpha)
  }
}

## 🔴 Critical: KL Scaling with Minibatches

This is probably your **single biggest training issue**, alongside Issue #4 from the previous review.

```r
kl <- model_fit$get_model_kld() / batch_size
loss <- mse + kl
```

`nnf_mse_loss` returns the **mean** over the batch by default, so `mse ≈ (1/B) Σ_batch (y - ŷ)²`. This is an unbiased estimate of the per-observation MSE regardless of batch size. But the KL term is a fixed penalty over the entire parameter space — it doesn't scale with the batch. The correct normalization is to divide by the **total training set size**, not the batch size:

$$\mathcal{L} = \frac{1}{B}\sum_{i \in \text{batch}} \ell(y_i, \hat{y}_i) + \frac{1}{N_{\text{train}}} \text{KL}(q \| p)$$

When `batch_size < N_train`, dividing by `batch_size` **overweights the KL** by a factor of `N_train / batch_size`. For example, with 1000 training observations and batch size 100, the prior penalty is **10× too strong**, which would heavily over-regularize and dramatically slow convergence. Fix:

```r
kl <- model_fit$get_model_kld() / ttsplit_ind
loss <- mse + kl
```

*(When `batch_size` is `NULL`, you set `batch_size <- ttsplit_ind`, so the current code happens to be correct in the full-batch case — but it breaks as soon as minibatching is enabled.)*

## 🟠 Convergence / Performance Issues

### 2. No KL Warm-Up / Annealing

From epoch 1, the KL is at full strength. In variational inference, this often causes the model to collapse toward the prior before the weights learn useful representations. A standard remedy is to linearly anneal the KL weight from 0 to 1 over an initial warm-up period:

```r
kl_weight <- min(1, epoch / kl_warmup_epochs)
loss <- mse + kl_weight * kl
```

A typical `kl_warmup_epochs` might be 10–20% of total training epochs.

### 3. Test/Plot Forward Passes Build Unnecessary Computation Graphs

During the reporting phase, the model is called without disabling gradient tracking:

```r
# Test MSE computation
yhat_test <- model_fit(x_test)
mse_test <- nnf_mse_loss(yhat_test, y_test)

# Plot prediction
plotdf$y <- as_array(model_fit(x_plot))
```

Each of these builds a full autograd graph (including sampling noise, exponentiation, linear layers, etc.) that is never used for backpropagation. This wastes significant memory — especially if `x_test` or `x_plot` are large — and can trigger unnecessary GPU memory allocation. Wrap these in `with_no_grad()`:

```r
with_no_grad({
  yhat_test <- model_fit(x_test)
  mse_test <- nnf_mse_loss(yhat_test, y_test)
})
```

### 4. Test MSE Uses Stochastic Forward Pass

Related to the above: the reporting-phase calls go through the stochastic `forward()`, which samples random weights and z values. This makes your monitored test MSE **noisy**, which can obscure the true convergence trend and make it harder to diagnose problems. Consider using `get_nn_mod_Ey()` (the deterministic point-estimate prediction) for monitoring:

```r
with_no_grad({
```

## 🟡 Logic Bugs

### 5. Plot Window `if/else` Ordering

```r

  start_plot_row_ind <- row_ind - 30
```

```r
  start_plot_row_ind <- row_ind - 10
```

### 6. First Batch Is Skipped on the First Pass


### 1. `clip_var = TRUE` — Type Error Waiting to Fire

Every layer is initialized with:

Inside `torch_hs`, the guard is `!is.null(self$clip_var)`, and the clamping uses `log(self$clip_var)`. In R, `log(TRUE) = log(1) = 0`, so if `clip_variances()` is ever called, **all log-variances get clamped to ≤ 0** (i.e., σ² ≤ 1). This is almost certainly not the intent.

Currently `clip_variances()` doesn't appear to be called anywhere in your training loop or forward pass, so this is dormant. But if you ever wire it in (as is common in Bayesian compression workflows), it will silently cap all weight variances at 1. Fix by passing a numeric threshold or `NULL`:

```r
clip_var = 10   # σ² ≤ 10, for example
# or
clip_var = NULL  # disable clamping
```

## 🟠 Convergence / Performance Issues

### 3. Implicit Noise Variance Mismatch

Your loss is:

```r
loss <- mse + kl   # where kl = KL / N_train
```

`nnf_mse_loss` computes `mean((y - ŷ)²)`. For a Gaussian likelihood with variance σ², the per-observation negative log-likelihood (up to a constant) is `(y - ŷ)² / (2σ²)`. So your loss implicitly assumes **σ² = 0.5**, but your data is generated with `err_sigma = 1` (σ² = 1). This over-weights the likelihood relative to the KL by a factor of 2, which means the posterior will be tighter (less regularized) than the true Bayesian posterior.

For a properly calibrated ELBO:

```r
sigma_sq <- 1  # match your data-generating process
loss <- mse / (2 * sigma_sq) + kl
```

Or equivalently, scale the KL:

```r
loss <- mse + 2 * kl   # since mse already divides by N
```

This is a common modeling choice that many practitioners leave as-is, but it matters for your horseshoe prior since the balance between likelihood and KL directly controls how aggressively the shrinkage operates.

### 4. Full-Batch Training at Scale

With `batch_size = NULL`, every gradient step processes all 1000 training observations. Combined with 300K epochs, that's 300 million observation-level forward/backward computations. Minibatching (e.g., batch size 100–200) would:

- Reduce wall-clock time per epoch
- Add beneficial stochastic noise that helps escape local optima in the ELBO landscape
- Allow more frequent parameter updates per pass through the data

Just remember to apply the KL scaling fix from the previous review (`kl / N_train` not `kl / batch_size`) when enabling minibatching.

### 5. Fixed Learning Rate for 300K Epochs

A constant `lr = 0.001` for the full run means the step size that works well early on (when gradients are large) may be too large late in training, causing oscillation around the optimum. Consider a scheduler:

```r
scheduler <- lr_step(optim_model_fit, step_size = 100000, gamma = 0.5)

# Inside training loop, after optim_model_fit$step():
scheduler$step()
```

This halves the learning rate every 100K epochs, allowing finer convergence in later stages.

### 6. Horseshoe Prior on the Output Layer

Your final layer `fc3` (32 → 1) uses the full horseshoe machinery. This means:

- A **single scalar output** goes through stochastic z-sampling, local reparameterization, and horseshoe shrinkage

- But it also adds an additional noise source to every prediction, increasing ELBO gradient variance

You: Would a cyclical learning rate scheduler work better than halving it every X epochs?
AI: Great question. The short answer is: **for variational inference with a horseshoe prior, cyclical LR is riskier than it would be in standard DNN training**, but a carefully designed variant could work. Here's the reasoning:

---

## Why Cyclical LR Is Attractive

The theoretical appeal is real. Your ELBO landscape is **multimodal** — the horseshoe prior creates two attracting regions for each input feature (shrink toward zero vs. remain large), and a cyclical schedule can help the optimizer explore these modes rather than committing too early. Standard step decay is purely exploitative: once the LR drops, you're locked into whatever basin you're in.

## Why It's Risky for This Model

Your model has **qualitatively different parameter types** that respond very differently to LR spikes:

| Parameter | Behavior under high LR |
|-----------|----------------------|
| `weight_mu` | Relatively robust — similar to standard DNN weights |
| `weight_logvar` | Sensitive — overshooting can cause variance to explode (`exp(logvar)`), creating NaN cascades |
| `sa_mu`, `sb_mu`, `atilde_mu`, `btilde_mu` | Very sensitive — these control `z = exp(...)`, so small perturbations get exponentiated |
| `sa_logvar`, etc. | Doubly sensitive — perturbations are exponentiated *twice* (once for variance, once for z) |

When a cyclical schedule pushes the LR back up, the shrinkage parameters can get kicked far from their current values. Because these pass through `exp()` in the forward pass, even modest overshooting in log-space translates to **huge** changes in the effective scale of z. This can undo epochs of progress in a single high-LR phase, and in the worst case, produce `Inf`/`NaN` values.

Additionally, if you combine cyclical LR with the KL warm-up I suggested earlier, the interaction becomes complex — the model would be simultaneously ramping up the KL penalty while the LR is cycling, which can create chaotic training dynamics.

## What I'd Recommend Instead

**Cosine annealing without restarts** gives you the best of both worlds — a smooth, gradual decay that spends more time at intermediate learning rates than step decay does, without the destabilizing upswings:

```r
# In R torch:
scheduler <- lr_cosine_annealing(optim_model_fit, T_max = sim_params$train_epochs)

# Inside training loop, after optim_model_fit$step():
scheduler$step()
```

This follows the schedule: $\eta_t = \eta_{\min} + \frac{1}{2}(\eta_{\max} - \eta_{\min})\left(1 + \cos\left(\frac{t}{T_{\max}} \pi\right)\right)$

which decays smoothly from your initial LR to near zero, spending proportionally more time at moderate rates than a step schedule would.

## If You Still Want to Try Cyclical

If you want the exploration benefits, there are two safer approaches:

### Option A: Cosine Annealing with Warm Restarts + Decaying Amplitude

Use cycles, but **shrink the amplitude** each cycle so late-training cycles are gentle:

```r
# Cosine annealing with warm restarts
scheduler <- lr_cosine_annealing(optim_model_fit, T_max = 50000)
# Call scheduler$step() each epoch; LR resets every 50K epochs
# Manually decay the base LR after each restart
```

### Option B: Differential Learning Rates

Use **parameter groups** with different LR schedules — stable (or lower) LR for the shrinkage and variance parameters, with a cyclical or higher LR only for the weight means:

```r
optim_model_fit <- optim_adam(
  list(
    list(params = weight_mu_params,    lr = 0.001),  # can cycle this
    list(params = logvar_params,       lr = 0.0005), # keep this stable
    list(params = shrinkage_params,    lr = 0.0005)  # keep this stable
  )
)
```

This lets the weight means explore freely while keeping the sensitive variational parameters on a monotonic decay. It does require extracting the parameters into separate groups, which is a bit more bookkeeping in R torch, but is probably the most principled approach.

---

## Summary

| Schedule | Exploration | Stability | Complexity | Recommendation |
|----------|:---------:|:---------:|:----------:|:--------------:|
| Step decay | Low | High | Low | Baseline |
| **Cosine annealing (no restarts)** | **Medium** | **High** | **Low** | **Best default** |
| Cyclical (fixed amplitude) | High | Low | Low | Avoid |
| Cyclical (decaying amplitude) | High | Medium | Medium | Viable |
| Differential LR groups | High | High | High | Best if feasible |

I'd start with plain cosine annealing — it's a one-line change, it's safe, and it almost always outperforms step decay for long training runs like yours. If convergence is still unsatisfactory after the other fixes, differential LR groups would be the next step.

## Schedule Options

```r
# 1. Linear ramp: simplest, most common
kl_weight_linear <- function(epoch, warmup_epochs) {
  min(1, epoch / warmup_epochs)
}

# 2. Sigmoid ramp: smoother transition near 0 and 1,
#    avoids the abrupt corner at the end of linear warm-up.
#    Reaches ~0.95 at warmup_epochs; ~1.0 shortly after.
kl_weight_sigmoid <- function(epoch, warmup_epochs) {
  midpoint <- warmup_epochs / 2
  steepness <- 10 / warmup_epochs   # controls how sharp the transition is
  1 / (1 + exp(-steepness * (epoch - midpoint)))
}

# 3. Cosine ramp: spends more time near 0 early on,
#    then accelerates toward 1. Reaches exactly 1 at warmup_epochs.
kl_weight_cosine <- function(epoch, warmup_epochs) {
  if (epoch >= warmup_epochs) return(1)
  0.5 * (1 - cos(pi * epoch / warmup_epochs))
}
```

I'd recommend **starting with linear** — it's the easiest to reason about and debug. Once everything else is working, you can experiment with the others.

## One Caveat

You: Hey, I realized that it may be a better idea to sample the network parameters several times and average over them when computing the ELBO.  What benefits does this provide (and potential downsides), and how can I implement this easily in my code?
AI: Great instinct — this is a well-known variance reduction technique in VI. Here's the full picture and two implementation options.

---

## Benefits

**1. Lower variance gradient estimates**

Your ELBO likelihood term is estimated with a single Monte Carlo sample per step. The gradient variance of this estimator scales as σ²/K where K is the number of samples. Going from K=1 to K=5 cuts gradient variance by ~5×. This is **on top of** the single-sample z fix from the first review — they address different noise sources:

| Noise source | Fix | What it reduces |
|---|---|---|
| 4 separate samples for z | Combine into 1 sample | Variance from redundant noise in z |
| 1 MC sample for the ELBO | Average K samples | Variance from the ELBO estimator itself |

**2. More reliable shrinkage decisions**

The horseshoe prior creates a bimodal landscape for each input — either shrink or don't. With a single noisy sample, the gradient can point toward "shrink" on one step and "don't shrink" on the next, causing oscillation. Multiple samples give clearer signal about which direction is truly downhill.

**3. Smoother KL–likelihood balance**

On any given step, a single-sample MSE could be anomalously high or low, temporarily distorting the balance with the KL term. Averaging smooths this out, which is particularly helpful during the KL warm-up phase when the balance is shifting.

---

## Downsides

**1. Linear increase in computation per step**

K samples means K forward passes. For your network (104→32→32→1), each forward pass is cheap, so K=5 roughly means ~5× longer per step.

**2. Diminishing returns**

Variance reduction is O(1/K), so the marginal value of each additional sample drops quickly. Going from 1→5 is transformative; going from 5→25 is barely noticeable.

**3. May be unnecessary after other fixes**

If you've already implemented the single-sample z fix and KL annealing, the gradient variance may already be low enough that K=1 suffices. I'd recommend getting those fixes working first, then experimenting with K>1 only if convergence is still unsatisfactory.

---

## Implementation

The key insight is that the **KL is computed analytically** — it doesn't involve sampling and doesn't need to be averaged. Only the likelihood (MSE) term needs multiple samples, since each forward pass draws different weight and z samples.

### Option A: Simple Averaging (cleaner code, more memory)

Accumulates K computation graphs in memory, then does a single backward pass:


### Option B: Gradient Accumulation (lower memory, slightly more complex)

Processes one sample at a time, calling `backward()` after each. Torch accumulates gradients automatically:

```r
## Replace the existing forward + loss block:
# ---- begin replacement ----
n_mc <- sim_params$n_mc_samples

optim_model_fit$zero_grad()

# KL: compute once, backward once
kl_raw <- model_fit$get_model_kld() / ttsplit_ind
kl_weight <- kl_weight_linear(epoch, sim_params$kl_warmup_epochs)
kl <- kl_weight * kl_raw
kl$backward()   # KL gradients stored in param$grad

# Likelihood: average over K samples via accumulated gradients
for (s in seq_len(n_mc)) {
  if (is.null(sim_params$batch_size)) {
    yhat_train <- model_fit(x_train)
    mse_s <- nnf_mse_loss(yhat_train, y_train) / n_mc
  } else {
    yhat_train <- model_fit(x_train[batch_inds, ])
    mse_s <- nnf_mse_loss(yhat_train, y_train[batch_inds]) / n_mc
  }
  mse_s$backward()  # gradients accumulate (add to existing KL grads)
}

optim_model_fit$step()

# For reporting (compute mse for display):
mse <- mse_s * n_mc  # last sample's mse, approximate; or track separately
# ---- end replacement ----
```

**Pros:** Only one computation graph in memory at a time — scales to any K and any model size.
**Cons:** Slightly more complex; the `mse` value for reporting is approximate (it's the last sample's value, not the true average). If you want exact reporting, accumulate it in a separate no-grad variable:

```r
# Inside the MC loop, before mse_s$backward():
mse_report <- if (s == 1) mse_s$item() else mse_report + mse_s$item()
# After loop:
mse_report <- mse_report  # already divided by n_mc via the / n_mc above
```

## Mode vs. Mean for the Point Estimate

`get_nn_mod_Ey` defaults to `ln_fcn = ln_mode`, which uses the **mode** of the log-normal distribution: `exp(μ - σ²)`. You also have `ln_mean` available, which uses the **mean**: `exp(μ + σ²/2)`. The distinction matters:

| | Mode: `exp(μ - σ²)` | Mean: `exp(μ + σ²/2)` |
|---|---|---|
| **Interpretation** | Most probable value of z | Expected value of z |
| **Effect of high variance** | Pushes z toward 0 (more aggressive shrinkage) | Pushes z away from 0 (less shrinkage) |
| **Better for** | Feature selection / sparsity assessment | Predictive accuracy |

For **monitoring test MSE during training**, I'd recommend `ln_mean`:

```r

  yhat_test <- get_nn_mod_Ey(model_fit, x_test, ln_fcn = ln_mean)
  mse_test <- nnf_mse_loss(yhat_test, y_test)
})
```

The reasoning: you want the test MSE to reflect the model's average predictive performance under the posterior, not its performance at the posterior mode. The mean gives `E[z]`, so when multiplied with `weight_mu` (which approximates `E[w̃]`), you get a better approximation of `E[z · w̃]` — at least when the covariance between z and w̃ is small, which is the case here since they're parameterized independently under q.

For **post-hoc feature selection** (kappas, dropout rates), the mode is more appropriate since it answers "what's the single most likely shrinkage state?" — and that's what your existing `get_kappas` functions already use.

## Add Two Methods to `torch_hs`

These compute point estimates of `z` from the log-normal variational parameters, keeping everything in torch tensors on the correct device:

```r
# Add inside the torch_hs nn_module definition:

get_z_mean = function() {
  # z = z_tilde * s, each a product of two independent log-normals.
  # Product of independent log-normals is log-normal:
  #   mu_z  = sum of component mus
  #   var_z = sum of component variances (NOT product!)
  # E[LogNormal] = exp(mu + var/2)
  mu_z  <- self$atilde_mu + self$btilde_mu + self$sa_mu + self$sb_mu
  var_z <- self$atilde_logvar$exp() + self$btilde_logvar$exp() +
           self$sa_logvar$exp()     + self$sb_logvar$exp()
  torch_exp(mu_z + var_z / 2)
},

get_z_mode = function() {
  # Mode[LogNormal] = exp(mu - var)
  mu_z  <- self$atilde_mu + self$btilde_mu + self$sa_mu + self$sb_mu
  var_z <- self$atilde_logvar$exp() + self$btilde_logvar$exp() +
           self$sa_logvar$exp()     + self$sb_logvar$exp()
  torch_exp(mu_z - var_z)
},
```

## Modify `forward` to Branch on Training Mode

Replace the existing `forward` method in `torch_hs`:

```r
forward = function(x) {
  if (self$training) {
    # ---- Stochastic forward pass (existing code) ----
    z_tilde <- self$reparameterize(self$atilde_mu, self$atilde_logvar)$mul(
      self$reparameterize(self$btilde_mu, self$btilde_logvar)
    )
    s <- self$reparameterize(self$sa_mu, self$sa_logvar)$mul(
      self$reparameterize(self$sb_mu, self$sb_logvar)
    )
    z <- z_tilde$mul(s)

    new_weight <- z * self$weight_mu
    new_var    <- z$pow(2) * self$weight_logvar$exp()

    act_mu  <- torch_mm(x, new_weight$t()) + self$bias_mu
    act_var <- torch_mm(x$pow(2), new_var$t()) + self$bias_logvar$exp()

    eps <- torch_randn_like(act_mu)
    return(act_mu + act_var$sqrt() * eps)

  } else {
    # ---- Deterministic forward pass ----
    z <- self$get_z_mean()   # swap to get_z_mode() for sparsity assessment
    new_weight <- z * self$weight_mu
    torch_mm(x, new_weight$t()) + self$bias_mu
  }
},
```

The key differences in the `else` branch:
- **z** is a point estimate, not a sample
- **Weights** use `weight_mu` directly (posterior mean), no variance term
- **Bias** uses `bias_mu` directly
- **No noise injection** — no `eps` sampling

## Mean vs. Mode — When to Use Each

The default in the `else` branch above uses `get_z_mean()`. Here's when you'd switch:

```r
# For predictive accuracy (test MSE, function plots):
z <- self$get_z_mean()     # E[z] — accounts for log-normal skew

# For sparsity assessment (kappas, feature selection):
z <- self$get_z_mode()     # mode[z] — "most likely" shrinkage state
```

The difference is most pronounced when the posterior variance is large (i.e., the model is uncertain about a feature). In that case, the mean is pulled upward by the log-normal's right skew, while the mode is pulled toward zero. In a standard linear regression y = Xβ + ε with the horseshoe prior, the global shrinkage parameter τ controls the **overall sparsity level** — how many coefficients are expected to be near zero vs. meaningfully nonzero.

- **p₀** = expected number of nonzero coefficients
- **n** = number of observations

2. **"Sparsity" is well-defined** — either β_j ≠ 0 (feature j matters) or β_j = 0 (it doesn't).
3. **p₀ is meaningful** — you can genuinely talk about "how many of the p features are relevant."
4. **The √n scaling** comes from the Fisher information in the Gaussian likelihood, reflecting how much data you have to overcome the prior.

## What Breaks in a Neural Network

In your implementation, `z` plays the role of λ·τ (local × global shrinkage), gating each input feature before it enters the network. The global component `s` (= s_a · s_b) corresponds to τ. But the assumptions behind the τ₀ formula no longer hold:

### 1. Features Don't Have Independent, Additive Effects

```
Linear:   y = β₁x₁ + β₂x₂ + β₃x₃
Neural:   y = W₂ · ReLU(W₁ · (z ⊙ x))
```

- **What counts as "a parameter"?** Is p the number of input features (where z acts)? Or the total number of weights across all layers? z only gates the input layer, but the second-layer weights are also doing important work.
- **What counts as "relevant"?** A feature might be irrelevant on its own but critical through an interaction (e.g., x₁·x₂). The horseshoe sees x₁ and x₂ individually and might try to shrink both, even though their product is the true signal.

### 3. The √n Scaling Doesn't Transfer

The 1/√n term in the τ₀ formula comes from the **Fisher information of the linear Gaussian model**. The information geometry of a neural network is fundamentally different:

- The Fisher information matrix is not diagonal (parameters interact)
- It depends on the current parameter values (the landscape changes during training)
- The effective number of parameters is generally much less than the literal count

So even if you knew p₀ exactly, dividing by √n could be too aggressive or too lenient depending on the network architecture and the true function.

### 4. The Horseshoe's Optimality Results Don't Apply

The theoretical appeal of the horseshoe — near-minimax posterior contraction rates, optimal handling of sparse signals — was proven for the **normal means problem** and extended to linear regression. These results rely on:

- Posterior concentration around the true sparse β
- The likelihood being log-concave in the parameters
- Each parameter having a "one-to-one" relationship with a signal component

None of these hold for neural networks, where the loss landscape is non-convex, the parameterization is non-identifiable (permutation symmetry), and the relationship between parameters and function output is highly nonlinear.

### 5. Global Shrinkage Interacts With Network Width

Consider your architecture: 104 inputs → 32 hidden → 32 hidden → 1 output. Each input feature connects to 32 first-layer hidden units. If the network were wider (say 256 hidden units), the same feature would have more "capacity" to express itself, meaning the same τ would shrink it relatively less in terms of functional effect. In linear regression, there's no analogous interaction between the number of coefficients and the capacity per coefficient.

## The Root Cause

The issue is **scale non-identifiability**. In a linear model y = Xβ + ε, the horseshoe prior on β directly mediates between the data and the predictions — if τ collapses, the fit suffers, and the likelihood pushes back. In your neural network, the gates z_j and the downstream weights W₁, W₂ can trade off: the network learns to shrink τ toward zero while compensating by amplifying later layers. The likelihood never "feels" the collapse because the network routes around it. This removes the tension that keeps τ properly calibrated in the linear case.

## Why Post-Training Correction Is the Right Instinct

Separating the learning phase (where the network discovers which features matter) from the inference phase (where you make statistically grounded selection decisions) is actually principled. It's conceptually similar to **projection predictive variable selection** (Piironen, Paasiniemi & Vehtari, 2020), where you first fit a rich reference model and then project onto sparse submodels for inference. Your trained network is the reference model; the post-training correction is the projection step.

## The Proposed Fix: Post-Training τ Recalibration

The idea is simple: **keep the learned local scale parameters λ_j, discard the collapsed τ, and solve for the τ\* that is self-consistent with the horseshoe's own implied sparsity.**

### The Self-Consistency Equation

From Piironen & Vehtari, the recommended global shrinkage for an expected p₀ relevant features out of p total is:

$$\tau_0 = \frac{p_0}{p - p_0} \cdot \frac{\hat{\sigma}}{\sqrt{n}}$$

Inverting this, a given τ\* implies an expected number of relevant features:

$$p_0(\tau^*) = \frac{p \cdot \tau^* \sqrt{n}}{\hat{\sigma} + \tau^* \sqrt{n}}$$

Meanwhile, the effective number of active features given τ\* and the learned λ_j values is:

$$m_{\text{eff}}(\tau^*) = \sum_{j=1}^{p} \left(1 - \kappa_j(\tau^*)\right) = \sum_{j=1}^{p} \frac{\tau^{*2} \lambda_j^2}{1 + \tau^{*2} \lambda_j^2}$$

The **self-consistency condition** requires these to agree:

$$\sum_{j=1}^{p} \frac{\tau^{*2} \lambda_j^2}{1 + \tau^{*2} \lambda_j^2} = \frac{p \cdot \tau^* \sqrt{n}}{\hat{\sigma} + \tau^* \sqrt{n}}$$

This is a single equation in one unknown (τ\*), easily solved by bisection.

### Why a Non-Trivial Solution Is Guaranteed to Exist

- **As τ\* → 0:** The left side behaves as τ\*² Σλ_j² (quadratic), while the right side behaves as pτ\*√n/σ̂ (linear). So the right side dominates: p₀ > m_eff.
- **As τ\* → ∞:** The left side approaches p as p − C/τ\*² while the right side approaches p as p − pσ̂/(τ\*√n). Since 1/τ\*² vanishes faster than 1/τ\*, we get m_eff > p₀.

Since the left goes from below to above the right, they must cross at least once by the intermediate value theorem. The crossing point is your τ\*.

### What You Compute

After finding τ\*, the corrected shrinkage factors are:

$$\kappa_j^* = \frac{1}{1 + \tau^{*2} \lambda_j^2}$$

For the **regularized** horseshoe (which you're using), substitute the regularized local scale:

$$\tilde{\lambda}_j^2 = \frac{c^2 \lambda_j^2}{c^2 + \tau^{*2} \lambda_j^2}$$

$$\tilde{\kappa}_j^* = \frac{1}{1 + \tau^{*2} \tilde{\lambda}_j^2}$$

Features with κ_j\* near 0 are selected (signal); features with κ_j\* near 1 are excluded (noise).

### Implementation

```python
import numpy as np
from scipy.optimize import brentq

def recalibrate_tau(lambda_j, sigma_hat, n, p, tau_range=(1e-10, 1e3)):
    """
    Find self-consistent tau* given learned local scales.
    
    Args:
        lambda_j: array of learned local scale parameters (posterior means)
        sigma_hat: residual std from trained network
        n: number of training observations
        p: number of input features
        tau_range: search interval for bisection
    
    Returns:
        tau_star: recalibrated global shrinkage
        kappa_star: corrected shrinkage factors
        m_eff: effective number of selected features
    """
    lam2 = lambda_j ** 2
    
    def equation(log_tau):
        tau = np.exp(log_tau)
        tau2 = tau ** 2
        # Left side: effective number of features
        m_eff = np.sum(tau2 * lam2 / (1 + tau2 * lam2))
        # Right side: implied p0 from P&V formula
        p0 = p * tau * np.sqrt(n) / (sigma_hat + tau * np.sqrt(n))
        return m_eff - p0
    
    log_tau_star = brentq(
        equation, 
        np.log(tau_range[0]), 
        np.log(tau_range[1])
    )
    tau_star = np.exp(log_tau_star)
    
    kappa_star = 1.0 / (1.0 + tau_star**2 * lam2)
    m_eff = np.sum(1 - kappa_star)
    
    return tau_star, kappa_star, m_eff
```

## The Critical Caveat: Are the Learned λ_j Reliable?

This is the key question that determines whether recalibration works. During training, the gradient signal for λ_j scales as:

$$\frac{\partial \mathcal{L}}{\partial \lambda_j} = \frac{\partial \mathcal{L}}{\partial z_j} \cdot \frac{2\tau^2 \lambda_j}{(1 + \tau^2 \lambda_j^2)^2}$$

When τ is very small, this gradient is proportional to τ², so learning of λ_j becomes extremely slow. The variational posterior for λ_j gets dominated by the KL term pulling it toward the half-Cauchy prior rather than being shaped by the data. In the worst case, the λ_j values carry almost no information about feature importance.

**How to diagnose this:** After training, check whether the distribution of learned λ_j values shows meaningful spread. If the λ_j values span several orders of magnitude, they're informative and recalibration should work well. If they're all clustered in a narrow range near the prior's typical values, they may be uninformative.

## A Note on Statistical Rigor

To be fully honest: the horseshoe's theoretical guarantees — posterior concentration rates, near-minimax optimality, the "no unnecessary shrinkage" property — are proven for linear models. In the neural network setting, we are borrowing the horseshoe's shrinkage structure and applying it to quantities (learned λ_j or gradient-based coefficients) that are themselves estimates from a complex nonlinear model. The post-training recalibration gives you something I'd describe as **horseshoe-calibrated variable selection applied to a neural network's learned importance structure** — substantially more principled than ad hoc thresholding or ranking, but not carrying the full theoretical weight of linear horseshoe inference.

## What the Frobenius Norm Gets Right

Your `get_kappas_frobcorrected` computes:

$$z_j^{\text{eff}} = z_{1,j} \cdot \prod_  $$z_j^{\text{eff}} = z_{1,j} \cdot \prod_{\ell=2}^{L} \|W_\ell \cdot \text{diag}(z_\ell)\|_F$$

This captures the right idea — downstream layers absorb signal that τ "loses" — but the Frobenius norm introduces systematic distortions.

## Three Problems with the Frobenius Norm

### 1. Dimension scaling

For a d×d matrix with unit singular values, ‖W‖_F = √d but the actual per-direction amplification is 1. So wider layers look like they amplify more, even when they don't. In your 32→32 layer with Kaiming initialization, ‖W₂‖_F ≈ √32 ≈ 5.7, while the spectral norm ≈ 1.

### 2. Composition overestimates

Your code multiplies Frobenius norms across layers:

```r

```

But ‖AB‖_F ≤ ‖A‖_F · ‖B‖_F, with equality only when the matrices are rank-1. Multiplying Frobenius norms gives an **upper bound** that can be extremely loose for full-rank matrices. The actual composed matrix norm is typically much smaller.

### 3. Uniform scaling across features

Frobenius norm produces one scalar per layer. But different input features route through different hidden units and experience different amplification. Feature j's signal travels through a specific direction in hidden space determined by W₁[:,j], and how much *that particular direction* gets amplified by downstream layers varies by feature.

## Better Alternatives (Best to Simplest)

### Option A: Per-Feature Effective Coefficients (Most Principled)

Compose the downstream matrices directly, then compute per-feature amplification:

```r
get_kappas_composed <- function(nn_mod, ln_fcn = ln_mean) {
  num_layers <- length(nn_mod$children)
  
  # Compose downstream matrix: M = W_L diag(z_L) ... W_2 diag(z_2)
  # Start from the output layer and work backward to layer 2
  M <- diag(sqrt(get_zsq(nn_mod$children[[num_layers]], ln_fcn)))
  M <- as.matrix(nn_mod$children[[num_layers]]$weight_mu) %*% M
  
  if (num_layers > 2) {
    for (l in (num_layers - 1):2) {
      z_l <- sqrt(get_zsq(nn_mod$children[[l]], ln_fcn))
      W_l <- as.matrix(nn_mod$children[[l]]$weight_mu)
      M <- M %*% W_l %*% diag(z_l)
    }
  }
  
  # M is now (d_out × d_hidden1)
  # Per-feature effective coefficient:
  #   beta_j^eff = z_{1,j} * || M %*% W_1[:,j] ||_2
  z1 <- sqrt(get_zsq(nn_mod$children[[1]], ln_fcn))
  W1 <- as.matrix(nn_mod$children[[1]]$weight_mu)
  
  # MW1 has shape (d_out × d_in); column j is M %*% W1[:,j]
  MW1 <- M %*% W1
  col_norms <- sqrt(colSums(MW1^2))  # ||M W1[:,j]||_2 for each j
  
  beta_eff <- z1 * col_norms
  kappas <- 1 / (1 + beta_eff^2)
  
  return(list(
    "kappas" = kappas,
    "beta_eff" = beta_eff,
    "M" = M
  ))
}
```

This computes the **actual** end-to-end effective coefficient for each input feature, accounting for exactly how that feature's signal propagates through the full network. No approximation beyond ignoring ReLU.

## ReLU: Does It Matter?

The composed matrix M ignores ReLU nonlinearities between layers. This introduces a multiplicative dampening (ReLU zeros out ~half the activations on average), which means M overestimates the true amplification by roughly 2^(L-2) for L layers. Two observations:

1. **It doesn't affect feature ranking.** The dampening is approximately uniform across features, so κ_j values maintain their relative ordering.
2. **The self-consistency recalibration absorbs it.** If you feed the effective coefficients β_j^eff into the τ* recalibration from the previous discussion, the self-consistency equation automatically compensates for uniform scaling errors.

So for variable selection purposes, ignoring ReLU is fine. If you wanted to account for it precisely, you'd compute the average Jacobian across training data, but that's considerably more expensive and unlikely to change your selection decisions.

The key insight is that κ and α are measuring different aspects of the **same posterior quantity**, and combining them gives you something neither provides alone.

---

## What κ and α Each Capture

**κ_j = 1/(1 + z_j²)** uses a point estimate of z_j (mode or mean). It answers: "given our best guess of z_j, how much is feature j shrunk?"

**α_j = exp(σ²_{log z}) − 1** uses only the variance of log(z_j). It answers: "how noisy is the gating variable for feature j?" This is what Louizos et al. threshold for pruning.

The problem is that **each ignores what the other captures**:

- κ alone ignores posterior uncertainty. Two features can have the same κ but very different reliability — one with a tight posterior, the other barely identified.
- α alone ignores posterior location. A feature with high α (noisy gate) could still have a very large μ_z, meaning it's probably signal despite the uncertainty. Conversely, a feature with low α (precise gate) that's precisely centered at zero is definitely noise.

---

## The Unification: Posterior Inclusion Probability

Since log(z_j) is normal under the variational posterior, you can compute the **exact posterior probability** that κ_j falls below any threshold t:

$$\kappa_j < t \iff z_j^2 > \frac{1-t}{t} \iff \log(z_j) > \frac{1}{2}\log\!\left(\frac{1-t}{t}\right)$$

Since log(z_j) ~ N(μ_z, σ²_z):

$$P(\kappa_j < t) = \Phi\!\left(\frac{\mu_z - \frac{1}{2}\log\!\left(\frac{1-t}{t}\right)}{\sigma_z}\right)$$

At the natural threshold **t = 0.5** (feature is more signal than noise), this simplifies beautifully:

$$\boxed{P_j \;\equiv\; P(\kappa_j < 0.5) \;=\; \Phi\!\left(\frac{\mu_z}{\sigma_z}\right)}$$

The inclusion probability is just the standard normal CDF evaluated at the **signal-to-noise ratio** of log(z_j).

---

## How This Connects to α

Since α = exp(σ²_z) − 1, we have σ_z = √(log(1 + α)), so:

$$P_j = \Phi\!\left(\frac{\mu_z}{\sqrt{\log(1 + \alpha)}}\right)$$

The behavior is intuitive:

| μ_z | α | P_j | Interpretation |
|---|---|---|---|
| Large positive | Small | ≈ 1 | Confidently signal |
| Large positive | Large | Moderate | Probably signal, but uncertain |
| Near zero | Small | ≈ 0.5 | Confidently irrelevant |
| Near zero | Large | ≈ 0.5 | Uncertain and irrelevant |
| Large negative | Any | ≈ 0 | Noise |

**Louizos et al.'s α-based pruning is equivalent to thresholding σ_z while ignoring μ_z.** This works for network compression (where you're asking "is this weight reliably nonzero?") but discards information that's critical for variable selection (where you need to distinguish "confidently small" from "uncertain").

## Integration with τ Recalibration and Downstream Correction

The variational parameters decompose μ_z into local and global components:

$$\mu_z = \underbrace{\frac{\mu_{\tilde\alpha_j} + \mu_{\tilde\beta_j}}{2}}_{\text{local: } \mu_{\text{local},j}} + \underbrace{\frac{\mu_{s_a} + \mu_{s_b}}{2}}_{\text{global: } \approx \log(\tau)}$$

When τ collapses, the global component drags all μ_z values negative, making all P_j ≈ 0. The recalibration from our previous discussion replaces the collapsed global component with the self-consistent τ\*. Additionally, the downstream amplification c_j from the composed weight matrix correction can be incorporated. All three corrections enter additively into the SNR:

$$P_j = \Phi\!\left(\frac{\mu_{\text{local},j} + \log(\tau^*) + \log(c_j)}{\sigma_{\text{local},j}}\right)$$

where:

- μ_local,j = (μ_ã_j + μ_b̃_j)/2 — learned local importance
- log(τ\*) — from self-consistency recalibration (replaces collapsed global)
- log(c_j) = log ‖M · W₁[:,j]‖₂ — downstream amplification for feature j
- σ²_local,j = (exp(logvar_ã_j) + exp(logvar_b̃_j))/4 — local uncertainty only (global uncertainty removed since τ\* is treated as fixed)

This is clean because all corrections are additive shifts in log-space, and the log-normal structure preserves normality throughout.

## The BFDR Connection — This Is the Key Payoff

Your existing BFDR framework plugs in directly with a proper Bayesian interpretation. Currently your code computes:

```r
BFDR <- function(dropout_probs, eta){
  delta_vec <- 1 * (dropout_probs <= eta)
  bfdr <- sum((dropout_probs) * delta_vec) / sum(delta_vec)
  ...
}
```

When `dropout_probs` contains α values, the BFDR calculation doesn't have a clean probabilistic meaning — α is a coefficient of variation, not a probability.

If you instead pass in **(1 − P_j)** as the dropout probabilities:

$$\text{BFDR} = \frac{\sum_j (1 - P_j) \cdot \delta_j}{\sum_j \delta_j}$$

This is now a **proper Bayesian FDR**: "of the features we selected, what is the average posterior probability that they are actually noise?" This is exactly the quantity that Bayesian FDR control is designed to bound.

Your existing `BFDR_eta_search` function already finds the threshold η that controls BFDR at a desired level, and `err_from_dropout` already computes FP/TP/FN/TN rates against the ground truth. **The entire evaluation pipeline works unchanged** — you just replace the input from α values to (1 − P_j) values.

## Implementation

```r
get_posterior_inclusion_prob <- function(
    nn_layer, 
    tau_star = NULL,        # recalibrated tau; NULL = use learned global
    c_j = NULL,             # downstream correction; NULL = no correction
    threshold = 0.5         # kappa threshold (0.5 = more signal than noise)
) {
  # Local parameters
  mu_local <- (as_array(nn_layer$atilde_mu) + 
               as_array(nn_layer$btilde_mu)) / 2
  var_local <- (exp(as_array(nn_layer$atilde_logvar)) + 
                exp(as_array(nn_layer$btilde_logvar))) / 4
  
  # Global component
  if (!is.null(tau_star)) {
    mu_global <- log(tau_star)
    var_global <- 0  # fixed after recalibration
  } else {
    mu_global <- (as_array(nn_layer$sa_mu) + 
                  as_array(nn_layer$sb_mu)) / 2
    var_global <- (exp(as_array(nn_layer$sa_logvar)) + 
                   exp(as_array(nn_layer$sb_logvar))) / 4
  }
  
  # Downstream correction
  log_c <- if (!is.null(c_j)) log(c_j) else 0
  
  # Combined parameters for log(z_j)
  mu_z  <- mu_local + mu_global + log_c
  sig_z <- sqrt(var_local + var_global)
  
  # P(kappa_j < threshold)
  log_odds <- 0.5 * log((1 - threshold) / threshold)  # 0 when threshold = 0.5
  P_j <- pnorm((mu_z - log_odds) / sig_z)
  
  # Dropout probability for BFDR
  dropout_prob <- 1 - P_j
  
  # Signal-to-noise ratio (for diagnostics)
  snr <- mu_z / sig_z
  
  return(list(
    "P_j" = P_j,
    "dropout_prob" = dropout_prob,
    "snr" = snr,
    "mu_z" = mu_z,
    "sig_z" = sig_z
  ))
}
```

## Caveats

**Variational underestimation of uncertainty:** Mean-field variational posteriors systematically underestimate σ_z (the posterior is too concentrated). This makes P_j values overconfident — closer to 0 or 1 than they should be. For clearly signal or clearly noise features, this doesn't matter. For borderline features, the inclusion probabilities should be treated as approximate. The BFDR may be slightly underestimated as a result.

**Corrupted local parameters:** As discussed previously, if τ collapsed severely during training, the local parameters μ_local,j may carry limited information. The SNR diagnostic helps detect this — if the spread of μ_z values across features is much smaller than σ_z, the local parameters aren't differentiating features well, and you should fall back to gradient-based effective coefficients.
